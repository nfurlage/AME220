2. Convert to hex format 

  rgb(213, 22,2)
    Work:
          r = 213 = 13*16 + 5*1 = d5
          g = 22 = 1*16 + 6*1 = 16
          b = 2 = 0*16 + 2*1 = 02
    Ans: #d51602

  rgb(112,226,52)
    Work:
          r = 112 = 7*16 + 0*1 = 70
          g = 226 = 14*16 + 2*1 = e2
          b = 52 = 3*16 + 4*1 = 34
    Ans: #70e234

  rgb(12,124,41)
    Work:
          r = 12 = 0*16 + 12*1 = 0c
          g = 124 = 7*16 + 12*1 = 7c
          b = 41 = 2*16 + 9*1 = 29
    Ans: #0c7c29

  rgb(223,32,113)
    Work:
          r = 223 = 13*16 + 15*1 = df
          g = 32 =  2*16 + 0*1 = 20
          b = 113 = 7*16 + 1*1 = 71
    Ans: #df2071

  rgba(112,23,124,.4)
    Work:
          r = 112 = 7*16 + 0*1 = 70
          g = 23 =  1*16 + 7*1 = 17
          b = 124 = 7*16 + 12*1 = 7c
    Ans: #70177c
